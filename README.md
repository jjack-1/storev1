# create table

```sql
create table store_tb
(
    id    integer generated by default as identity,
    price integer,
    stock integer,
    name  varchar(255),
    primary key (id)
);

create table log_tb
(
    id          integer generated by default as identity,
    qty         integer(255),
    store_id    integer,
    buyer       varchar(255),
    total_price integer,
    primary key (id)
)
```
